Predicate.hasGroup

Predicate

Given a tag group, evaluates to true if one or more of the tags on the file is in the group; otherwise, evaluates to false.

Examples

Tags: tag1, tag2, tag3

Group1: tag2, tag4

Group2: tag5, tag6

  • hasGroup Group1: True

  • hasGroup Group2: False

Usage

let Autotagical = https://siriusstarr.github.io/autotagical/dhall/package.dhall

in  Autotagical.Predicate.hasGroup
      ( Autotagical.Tags.group
          [ Autotagical.Tags.tag "tag1", Autotagical.Tags.tag "tag2" ]
      )

let hasGroup
    : TagGroup → Predicate
    = λ(group : TagGroup) →
      λ(Predicate : Type) →
        let PredicateF =
              < AlwaysF
              | BetweenF :
                  { lowerBoundF : Bound
                  , upperBoundF : Bound
                  , predicatesF : List Predicate
                  }
              | HasGroupF : TagGroup
              | HasTagF : { tagF : Tag, withValueF : Optional (List Text) }
              | NotF : Predicate
              >

        in  λ(Fix : PredicateF → Predicate) → Fix (PredicateF.HasGroupF group)

in  hasGroup
