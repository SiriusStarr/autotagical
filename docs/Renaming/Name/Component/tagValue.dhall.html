Renaming.Name.Component.tagValue

Component

Given a tag, evaluates to the value of the tag, if it is a tagged value, or to simply the tag itself if it is not a tagged value.

Important: This should always be wrapped with a predicate that confirms the tag is present on the file (whether the renaming rule itself has the predicate or placing it inside a conditional component). A warning will be logged and the file will fail renaming if the tag value of a tag not on a file is requested.

Examples

  • Tags: [ tag1=val1, tag2 ]

    Component: tagValue "tag1"

    Output: "val1"

  • Tags: [ tag1, tag2 ]

    Component: tagValue "tag1"

    Output: "tag1"

Usage

let Autotagical = https://siriusstarr.github.io/autotagical/dhall/package.dhall

in  Autotagical.Renaming.Name.Component.tagValue (Autotagical.Tags.tag "tag1")

let tagValue
    : Tag → Component
    = λ(t : Tag) →
      λ(Component : Type) →
        let ComponentF =
              < FileDuplicateNumberF
              | FileFormatAsF :
                  { componentsF : List Component, formatCaseF : Case }
              | FileIfDuplicateF : List Component
              | FileIfThenElseF :
                  { predicateF : Predicate
                  , trueComponentsF : List Component
                  , falseComponentsF : List Component
                  }
              | FileInterpretF :
                  { componentF : Component, interpretAsF : Interpret }
              | FileOriginalNameF
              | FileTagValueF : Tag
              | FileTextLiteralF : Text
              >

        in  λ(Fix : ComponentF → Component) → Fix (ComponentF.FileTagValueF t)
