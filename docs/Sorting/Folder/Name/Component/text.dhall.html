Sorting.Folder.Name.Component.text

Component

Given nonempty text, evaluates to that literal text. This is the most fundamental component in a name template.

Usage

let Autotagical = https://siriusstarr.github.io/autotagical/dhall/package.dhall

in  Autotagical.Sorting.Folder.Name.Component.text "A literal text string."

let text
    : Text → Component
    = λ(text : Text) →
      λ(Component : Type) →
        let ComponentF =
              < FolderFormatAsF :
                  { componentsF : List Component, formatCaseF : Case }
              | FolderIfThenElseF :
                  { predicateF : Predicate
                  , trueComponentsF : List Component
                  , falseComponentsF : List Component
                  }
              | FolderInterpretF :
                  { componentF : Component, interpretAsF : Interpret }
              | FolderOriginalNameF
              | FolderTagValueF : Tag
              | FolderTextLiteralF : Text
              >

        in  λ(Fix : ComponentF → Component) →
              Fix (ComponentF.FolderTextLiteralF text)

in  text
